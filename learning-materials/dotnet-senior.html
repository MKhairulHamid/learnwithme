// 1. Service per Business Capability // UserService - handles user management public class UserService { private readonly IUserRepository _userRepository; private readonly IMessageBus _messageBus; public async Task<User> CreateUserAsync(CreateUserRequest request) { var user = new User(request.Email, request.Name); await _userRepository.SaveAsync(user); // Publish domain event await _messageBus.PublishAsync(new UserCreatedEvent { UserId = user.Id, Email = user.Email, CreatedAt = DateTime.UtcNow }); return user; } } // 2. API Gateway Pattern public class ApiGateway { private readonly IHttpClientFactory _httpClientFactory; public async Task<UserProfileResponse> GetUserProfileAsync(int userId) { var userClient = _httpClientFactory.CreateClient("UserService"); var orderClient = _httpClientFactory.CreateClient("OrderService"); // Parallel calls to different services var userTask = userClient.GetAsync($"/users/{userId}"); var ordersTask = orderClient.GetAsync($"/users/{userId}/orders"); await Task.WhenAll(userTask, ordersTask); var user = await userTask.Result.Content.ReadFromJsonAsync<User>(); var orders = await ordersTask.Result.Content.ReadFromJsonAsync<List<Order>>(); return new UserProfileResponse { User = user, Orders = orders, TotalOrders = orders.Count }; } } // 3. Architecture Considerations:

  • Database per Service: Each service owns its data
  • Event-Driven Communication: Async messaging for loose coupling
  • Circuit Breaker Pattern: Handle service failures gracefully
  • Distributed Tracing: Monitor requests across services
  • API Versioning: Maintain backward compatibility

3. How do you implement Event Sourcing and CQRS patterns?

Event Sourcing: Store all changes as a sequence of events.

// Event Store public interface IEventStore { Task SaveEventsAsync(Guid streamId, IEnumerable<DomainEvent> events, int expectedVersion); Task<IEnumerable<DomainEvent>> GetEventsAsync(Guid streamId); } // Aggregate Base public abstract class AggregateRoot { public Guid Id { get; protected set; } public int Version { get; private set; } private readonly List<DomainEvent> _uncommittedEvents = new(); public IReadOnlyList<DomainEvent> UncommittedEvents => _uncommittedEvents.AsReadOnly(); protected void RaiseEvent(DomainEvent domainEvent) { _uncommittedEvents.Add(domainEvent); Apply(domainEvent); Version++; } public void LoadFromHistory(IEnumerable<DomainEvent> events) { foreach (var evt in events) { Apply(evt); Version++; } } public void MarkEventsAsCommitted() { _uncommittedEvents.Clear(); } protected abstract void Apply(DomainEvent domainEvent); } // Account Aggregate with Event Sourcing public class Account : AggregateRoot { public decimal Balance { get; private set; } public string AccountNumber { get; private set; } public Account() { } // For reconstruction public Account(string accountNumber, decimal initialBalance) { RaiseEvent(new AccountCreatedEvent { AggregateId = Guid.NewGuid(), AccountNumber = accountNumber, InitialBalance = initialBalance, Timestamp = DateTime.UtcNow }); } public void Deposit(decimal amount) { if (amount <= 0) throw new ArgumentException("Amount must be positive"); RaiseEvent(new MoneyDepositedEvent { AggregateId = Id, Amount = amount, Timestamp = DateTime.UtcNow }); } public void Withdraw(decimal amount) { if (amount <= 0) throw new ArgumentException("Amount must be positive"); if (Balance < amount) throw new InvalidOperationException("Insufficient funds"); RaiseEvent(new MoneyWithdrawnEvent { AggregateId = Id, Amount = amount, Timestamp = DateTime.UtcNow }); } protected override void Apply(DomainEvent domainEvent) { switch (domainEvent) { case AccountCreatedEvent created: Id = created.AggregateId; AccountNumber = created.AccountNumber; Balance = created.InitialBalance; break; case MoneyDepositedEvent deposited: Balance += deposited.Amount; break; case MoneyWithdrawnEvent withdrawn: Balance -= withdrawn.Amount; break; } } } // CQRS - Separate Read Models public class AccountSummaryReadModel { public Guid AccountId { get; set; } public string AccountNumber { get; set; } public decimal CurrentBalance { get; set; } public int TransactionCount { get; set; } public DateTime LastActivity { get; set; } } // Query Handler public class GetAccountSummaryHandler : IRequestHandler<GetAccountSummaryQuery, AccountSummaryReadModel> { private readonly IReadModelRepository _repository; public async Task<AccountSummaryReadModel> Handle(GetAccountSummaryQuery query) { return await _repository.GetAccountSummaryAsync(query.AccountId); } } // Event Projections public class AccountSummaryProjection : IEventHandler<MoneyDepositedEvent>, IEventHandler<MoneyWithdrawnEvent> { private readonly IReadModelRepository _repository; public async Task Handle(MoneyDepositedEvent evt) { var summary = await _repository.GetAccountSummaryAsync(evt.AggregateId); summary.CurrentBalance += evt.Amount; summary.TransactionCount++; summary.LastActivity = evt.Timestamp; await _repository.SaveAccountSummaryAsync(summary); } }
